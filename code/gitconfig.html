Git configuration

Source: code/gitconfig.cmd
First created: 2020-10-04
Last modified: 2021-05-07

Include the following Git aliases in .gitconfig for fewer keystrokes in your workflow: 

[alias]
  a = add
  aa = add --all
  ch = checkout
  chb = checkout -b
  c = commit
  cm = commit -m
  d = diff
  dw = diff --word-diff
  dd = diff --word-diff-regex=.
  ds = diff --staged
  dsw = diff --staged --word-diff
  dsd = diff --staged --word-diff-regex=.
  f = fetch
  l = log
  m = merge
  mn = merge --no-ff --no-commit
  p = push
  sh = show
  shw = show --word-diff
  shd = show --word-diff-regex=.
  shn = show --name-status
  shf = show --first-parent
  s = status
